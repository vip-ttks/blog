linux hook动态库

工作技术 admin 3个月前 (12-07) 15℃

需要hook动态库的时候，可以用如下方法，确定是必须要hook所有的函数

#include <stdio.h>
#include <stdlib.h>
#include <dlfcn.h>
#include <openssl/rsa.h>
static void* lib_handle = NULL;
__attribute__((constructor)) void before_main() { //init handle
  lib_handle = dlopen("/usr/lib64/libssl.so.10.old",RTLD_NOW);
} 
__attribute__((destructor)) void after_main() { //close handle
   dlclose(lib_handle);
} 

int RSA_private_encrypt(int flen, unsigned char *from,unsigned char *to, RSA *rsa, int padding)
{
	printf("---------is me---------\n");
	int (*fun_RSA_private_encrypt)(int flen, unsigned char *from,unsigned char *to, RSA *rsa, int padding);
	fun_RSA_private_encrypt=dlsym(lib_handle,"RSA_private_encrypt");
	return fun_RSA_private_encrypt(flen,from,to,rsa,padding);
}

直接执行g++编译

g++ -shared -fPIC -o libssl.so.10 -ldl
